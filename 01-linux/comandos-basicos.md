# Comandos básicos de linux 

pwd - muestra donde estoy
ls - lista de archivos
ls -la  - lista todo incluyendo ocultos y detalles 
cd - navegar a las carpetas
cd ~ - ir a home
cd .. - retroceder o subir un nivel en las ramificaciones de las carpetas
mkdir - crea una carpeta 
touch - crea un archivo vacio
nano - abre el editor de texto en la terminal
